Details: Driver (DBI)

DBIx::Introspector::Driver->new({
   name => 'DBI',
   connected_determination_strategy => sub { $_[1]->{Driver}{Name} },
   unconnected_determination_strategy => sub {
      my $dsn = $_[1] || $ENV{DBI_DSN} || '';
      my ($driver) = $dsn =~ /dbi:([^:]+):/i;
      $driver ||= $ENV{DBI_DRIVER};
      return $driver
   },
})
